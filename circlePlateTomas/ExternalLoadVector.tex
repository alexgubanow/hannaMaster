\section*{External load vector}
\begin{equation}\label{eqn:flexMatrix}
    {F_k}=
    \frac{2\pi b_k\rho_k}{3}
    \left\{\begin{matrix}
        3\rho_{k,2}-b_k \\[2ex]
        3\rho_{k,2}+b_k \\[2ex]
    \end{matrix}\right\}
    =
    \{\eta_k\}\rho_k
\end{equation}
